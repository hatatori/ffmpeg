Images to video (timelapse)

ffmpeg -i imgs/%d.png output.mp4
ffmpeg -i imgs/%d.png output.mp4
ffmpeg -i imgs/%d.png -vf fps=30 output.mp4
ffmpeg -framerate 60 -i imgs/%d.png output.mp4
ffmpeg -framerate 60 -i imgs/%d.png -r 30 output.mp4 # -r (set fps)
ffmpeg -framerate 60 -i imgs/%d.png -c:v libx264 output.mp4
ffmpeg -framerate 60 -i imgs/%d.png -c:v libx265 output.mp4
